    class Meta:
        model = get_user_model()
        fields = ('email', 'first_name', 'last_name',)

    class Meta(UserCreationForm.Meta):

        form = CustomUserCreationForm(request.POST)

            user = form.save()

        form = CustomUserCreationForm()
